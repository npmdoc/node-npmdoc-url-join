    url-join = function () {
  var input = arguments;
  var options = {};

  if (typeof arguments[0] === 'object') {
    // new syntax with array and options
    input = arguments[0];
    options = arguments[1] || {};
  }

  var joined = [].slice.call(input, 0).join('/');
  return normalize(joined, options);
}
